Task 01. Дан массив a1 = [4,5,6,7,12,34,56,78,90,11] - с помощью forEach переберите массив и создайте новый массив a1_res куда добавьте элементы данного массива умноженные на 2. Действия должны запускаться при вызове функции t1.

function t1() {
let a1 = [4, 5, 6, 7, 12, 34, 56, 78, 90, 11];
a1.forEach(elem => {
a1_res.push(elem * 2);
});
document.querySelector('.out-1').innerHTML = a1_res;
}

document.querySelector('.b-1').onclick = t1;

Task 09. Дан массив a9 = [[hi, mahai], [test, best]] - преобразуйте его в строку. Разделитель - дефис. Результат присвойте a9_res. Запускаться решение должно при вызове функции t9. Допускается лишний дефис в конце строки!!!

function t9() {
let a9 = [
["hi", "mahai"],
["test", "best"],
];
let out = document.querySelector(".out-9");

a9_res = [].concat(...a9).join("-");
out.textContent = a9_res;
}
document.querySelector(".b-9").onclick = t9;

Task 10. Дан массив a10 = {name: ivan, age: 15, sex: 1, id: 45} - преобразуйте его в GET строку (GET параметры). Найдите описание что такое GET строка самостоятельно. Разделитель - амперсанд. Результат присвойте a10_res. Запускаться решение должно при вызове функции t10. Допускается лишний амперсанд в конце строки!!!

function t10() {
let a10 = { name: "ivan", age: 15, sex: 1, id: 45 };
let out = document.querySelector(".out-10");

for (let key in a10) {
a10_res += `${key}=${a10[key]}&`;
}
out.textContent = a10_res;
}
document.querySelector(".b-10").onclick = t10;
